CSS – Grid Layout

<body>
	  	  <header>Header</header>
	  	  <main>Lorem ipsum…</main>
	  	  <nav>Navigation</nav>
		  <aside>Commercials</aside>

body {
	display: grid;
	grid: "head head head" 5em
	      "nav  main ads"
	      "foot foot foot" 2em
	      / 1fr  auto 1fr;
	min-height: 100vh;
}
header { grid-area: head }
footer { grid-area: foot }
main   { grid-area: main }
nav    { grid-area: nav }
aside  { grid-area: ads }
